## LANGUAGE SWITCHER TEMPLATE

```html
<!-- Language switcher buttons in nav -->
<div class="lang-switcher flex gap-1">
  <button onclick="setLang('hr')" class="lang-btn" data-lang-btn="hr">HR</button>
  <button onclick="setLang('en')" class="lang-btn" data-lang-btn="en">EN</button>
  <button onclick="setLang('de')" class="lang-btn" data-lang-btn="de">DE</button>
  <button onclick="setLang('it')" class="lang-btn" data-lang-btn="it">IT</button>
</div>

<!-- Usage for translatable text: -->
<span data-lang="hr">Hrvatski tekst</span>
<span data-lang="en" class="hidden">English text</span>
<span data-lang="de" class="hidden">German text</span>
<span data-lang="it" class="hidden">Italian text</span>

<!-- Language switch script (place before closing body) -->
<script>
function setLang(lang) {
  document.querySelectorAll('[data-lang]').forEach(el => {
    if (el.closest('[data-lang-group]')) {
      const group = el.closest('[data-lang-group]');
      group.querySelectorAll('[data-lang]').forEach(child => {
        child.classList.toggle('hidden', child.dataset.lang !== lang);
      });
    } else {
      el.classList.toggle('hidden', el.dataset.lang !== lang);
    }
  });
  document.querySelectorAll('[data-lang-btn]').forEach(btn => {
    btn.classList.toggle('active', btn.dataset.langBtn === lang);
  });
  document.documentElement.lang = lang;
  localStorage.setItem('sidro-lang', lang);
}
const savedLang = localStorage.getItem('sidro-lang') || 'hr';
document.addEventListener('DOMContentLoaded', () => setLang(savedLang));
</script>
```
